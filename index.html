    function updateClock() {
      const now = new Date();
      const time = now.toLocaleTimeString('pt-BR');
      const date = now.toLocaleDateString('pt-BR', { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' });
      document.getElementById('time').textContent = time;
      document.getElementById('date').textContent = date;

    setInterval(updateClock, 1000);

html
<!DOCTYPE html>
<html lang="pt-BR">
<head>
  <meta charset="UTF-8">
  <title>Clima ao Vivo</title>
  <meta name="viewport" content="width=1152, initial-scale=1.0">
  <style>
    body {
      margin: 0;
      background-color: #000;
      color: #fff;
      font-family: Arial, sans-serif;
      display: flex;
      flex-direction: column;
      align-items: center;
      justify-content: center;
      width: 1152px;
      height: 768px;
      overflow: hidden;
    }

    #weather {
      text-align: center;
    }

    #icon {
      width: 100px;
      height: 100px;
    }

    #temp {
      font-size: 80px;
      margin: 20px 0;
    }

    #desc {
      font-size: 40px;
      text-transform: capitalize;
    }

    #city {
      font-size: 30px;
      margin-top: 10px;
    }
  </style>
</head>
<body>

<div id="weather">
  <img id="icon" src="" alt="Ícone do clima">
  <div id="temp">--°C</div>
  <div id="desc">Carregando...</div>
  <div id="city">Sua Cidade</div>
</div>

<script>
  const apiKey = 'SUA_CHAVE_AQUI';
  const cidade = 'Sao Paulo'; // Altere para sua cidade

  async function getWeather() {
    const url = `https://api.openweathermap.org/data/2.5/weather?q=${cidade}&appid=${apiKey}&units=metric&lang=pt_br`;
    const res = await fetch(url);
    const data = await res.json();

    document.getElementById('temp').textContent = `${Math.round(data.main.temp)}°C`;
    document.getElementById('desc').textContent = data.weather[0].description;
    document.getElementById('city').textContent = data.name;
    document.getElementById('icon').src = `https://openweathermap.org/img/wn/${data.weather[0].icon}@2x.png`;
  }

  getWeather();
  setInterval(getWeather, 10 * 60 * 1000); // Atualiza a cada 10 minutos
</script>

</body>
</html>

    ];

    const diasSemana = ['Dom', 'Seg', 'Ter', 'Qua', 'Qui', 'Sex', 'Sáb'];

    function fetchPrevisao(cidade) {
      const url = `https://api.openweathermap.org/data/3.0/onecall?lat=${cidade.lat}&lon=${cidade.lon}&exclude=current,minutely,hourly,alerts&units=metric&lang=pt_br&appid=${apiKey}`;
      fetch(url)
        .then(res => res.json())
        .then(data => {
          const container = document.getElementById(`dias-${cidade.nome}`);
          container.innerHTML = '';

          data.daily.slice(0, 7).forEach((diaData, i) => {
            const dataObj = new Date(diaData.dt * 1000);
            const diaSemana = diasSemana[dataObj.getDay()];
            const tempMax = Math.round(diaData.temp.max);
            const tempMin = Math.round(diaData.temp.min);
            const chuva = diaData.rain ? `${diaData.rain.toFixed(1)} mm` : '0 mm';
            const icone = diaData.weather[0].icon;

            const bloco = document.createElement("div");
            bloco.classList.add("dia");
            bloco.innerHTML = `
              <div>${diaSemana}</div>
              <img src="https://openweathermap.org/img/wn/${icone}.png" alt="clima">
              <div class="temp">${tempMax}° / ${tempMin}°</div>
              <div class="chuva">${chuva}</div>
            `;
            container.appendChild(bloco);
          });
        })
        .catch(err => {
          console.error(`Erro ao buscar dados de ${cidade.nome}:`, err);
        });
    }

    cidades.forEach(c => fetchPrevisao(c));

    // Atualiza a cada 3 horas (em milissegundos)
    setInterval(() => {
      cidades.forEach(c => fetchPrevisao(c));
    }, 3 * 60 * 60 * 1000);
